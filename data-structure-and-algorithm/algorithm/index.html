    3. 排序算法

    3.1. 分类

    • 内部排序：将需要处理的所有数据都加载到内部存储器（内存）中进行排序。
      • 插入排序：直接插入排序，希尔排序。
      • 选择排序：简单选择排序，堆排序。
      • 交换排序：冒泡排序，快速排序。
      • 归并排序。
      • 基数排序。
    • 外部排序：数据量过大，无法全部加载到内存中，需要借助外部存储（文件等）进行排序。

    3.2. 冒泡排序

    3.2.1. 基本思想

    依次比较相邻元素的值，若发现逆序则交换。

    3.2.2. 代码实现

    public class BubbleSort {
    public static void sort(int[] nums) {
        for (int i = 0; i < nums.length - 1; i++) {
            boolean flag = false;
            for (int j = 0; j < nums.length - i - 1; j++) {
                if (nums[j] > nums[j + 1]) {
                    flag = true;
                    int tmp = nums[j];
                    nums[j] = nums[j + 1];
                    nums[j + 1] = tmp;
                }
            }
            // 如果某一趟比较下来未发生交换则说明序列有序
            if (!flag) {
                break;
            }
        }
    }
}

    3.3. 选择排序

    3.3.1. 基本思想

    第 1 次从 nums[0]到 nums[n-1]从选取最小值与 nums[0]进行交换，第二次从 nums[1]到 nums[n-1]从选取最小值与 nums[1]进行交换，…，经过 n-1 次，得到有序序列。

    3.3.2. 代码实现

    public class SelectSort {
    public static void sort(int[] nums) {
        for (int i = 0; i < nums.length - 1; i++) {
            int min = nums[i];
            int minIdx = i;
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j] < min) {
                    min = nums[j];
                    minIdx = j;
                }
            }
            // 如果当前位置不是最小值，则交换
            if (minIdx != i) {
                nums[minIdx] = nums[i];
                nums[i] = min;
            }
        }
    }
}

    4. 查找算法

    4.1. 顺序（线性）查找

    public static int search(int[] nums, int num) {
    for (int i = 0; i < nums.length; i++) {
        if (nums[i] == num) {
            return i;
        }
    }
    return -1;
}

    4.2. 二分查找

    • 递归实现。
    public static int search(int[] nums, int num, int l, int r) {
    if (l > r) {
        return -1;
    }
    int mid = (l + r) / 2;
    if (nums[mid] == num) {
        return mid;
    } else if (nums[mid] > num) {
        return search(nums, num, l, mid - 1);
    } else {
        return search(nums, num, mid + 1, r);
    }
}

public static List<Integer> searchMult(int[] nums, int num, int l, int r) {
    List<Integer> list = new ArrayList<>();
    if (l > r) {
        return list;
    }
    int mid = (l + r) / 2;
    if (nums[mid] == num) {
        list.add(mid);
        int tmp = mid - 1;
        while (tmp > 0 && nums[tmp] == num) {
            list.add(tmp--);
        }
        tmp = mid + 1;
        while (tmp < nums.length && nums[tmp] == num) {
            list.add(tmp++);
        }
        return list;
    } else if (nums[mid] > num) {
        return searchMult(nums, num, l, mid - 1);
    } else {
        return searchMult(nums, num, mid + 1, r);
    }
}
    • 非递归实现。
    public static int search(int[] nums, int num) {
    int l = 0, r = nums.length - 1;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (nums[mid] == num) {
            return mid;
        } else if (nums[mid] > num) {
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return -1;
}

    4.3. 插值查找

    数据量较大，关键字分布比较均匀，插值查找速度较快。

    public static int search(int[] nums, int num) {
    int l = 0, r = nums.length - 1;
    while (l <= r) {
        // (num - nums[l]) / (mid - l) = (nums[r] - nums[l]) / (r - l)
        // (num - nums[l]) * (r - l) / (nums[r] - nums[l]) + l
        int mid = (num - nums[l]) * (r - l) / (nums[r] - nums[l]) + l;
        if (nums[mid] == num) {
            return mid;
        } else if (nums[mid] > num) {
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return -1;
}

    4.4. 斐波那契查找

    public static int search(int[] nums, int num) {
    // f(n+2) -1 = f(n) -1 + 1 + f(n+1)-1
    int low = 0, high = nums.length - 1;
    int[] f = f();
    int k = 0;
    while (high > f[k] - 1) {
        k++;
    }
    int[] copy = Arrays.copyOf(nums, f[k]);
    for (int i = high + 1; i < copy.length; i++) {
        copy[i] = nums[high];
    }
    while (low <= high) {
        int mid = low + f[k - 1] - 1;
        if (copy[mid] == num) {
            return mid > high ? high : mid;
        } else if (copy[mid] > num) {
            high = mid - 1;
            k = k - 1;
        } else {
            low = mid + 1;
            k = k - 2;
        }
    }
    return -1;
}

private static int[] f() {
    int[] f = new int[10];
    f[0] = 1;
    f[1] = 1;
    for (int i = 2; i < f.length; i++) {
        f[i] = f[i - 1] + f[i - 2];
    }
    return f;
}
