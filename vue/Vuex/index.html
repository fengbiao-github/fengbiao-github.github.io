    1. 基本概念

    1.1. 简介

    Vuex 是专门在 Vue 中实现集中式状态（数据）管理的一个 Vue 插件，对 vue 应用中多个组件的共享状态进行集中式管理（读/写），也是一种组件间通信的方式，且适用于任意组件间通信。

    1.2. 使用场景

    • 多个组件依赖于同一状态。
    • 来自不同组件的行为需要变更同一状态。

    2. 环境搭建

    1. 安装 vuex。

      # 查看所有版本号。
npm view vuex versions
# vue2要求vuex版本必须是3.x
npm i vuex@3.6.2

    2. 新建仓库。

      store/index.js
      import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

// 响应组件中的动作
const actions = {}
// 操作数据
const mutations = {}
// 存储数据
const state = {}

// 创建并暴露store
export default new Vuex.Store({
  actions,
  mutations,
  state
})

    3. 将仓库挂载到 Vue 实例上。

      main.js
      import Vue from 'vue'
import App from './App.vue'

Vue.config.productionTip = false

// 如果在main.js中使用则会报错，即使在`import store from './store'`之前使用。
Vue.use(Vuex)

import store from './store'

new Vue({
  render: h => h(App),
  store
}).$mount('#app')

    3. 基本使用

    store/index.js
    import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

// 响应组件中的动作
const actions = {
  jiaWhenJi(context, value) {
    console.log('actions中jiaWhenJi被调用')
    if (context.state.sum % 2) {
      context.commit('JIA', value)
    }
  },
  jiaWait(context, value) {
    console.log('actions中jiaWait被调用')
    // 异步处理。请求服务端接口建议在此处编写，而非mutations，见工作原理。
    setTimeout(() => {
      context.commit('JIA', value)
    }, 1000)
  }
}
// 操作数据
const mutations = {
  JIA(state, value) {
    console.log('mutations中JIA被调用')
    state.sum += value
  },
  JIAN(state, value) {
    console.log('mutations中JIAN被调用')
    state.sum -= value
  }
}
// 存储数据
const state = {
  sum: 0
}

// 创建并暴露store
export default new Vuex.Store({
  actions,
  mutations,
  state
})
    MyCount.vue
    <template>
  <div>
    <h1>sum: {{ $store.state.sum }}</h1>
    <select v-model.number="n">
      <option value="1">1</option>
      <option value="2">2</option>
      <option value="3">3</option>
    </select>
    <button @click="inc">加</button>
    <button @click="des">减</button>
    <button @click="incWhenJi">当sum为奇数时加</button>
    <button @click="incWait">等等再加</button>
  </div>
</template>
<script>
  export default {
    name: 'MyCount',
    data() {
      return {
        n: 1
      }
    },
    mounted() {
      console.log(this)
    },
    methods: {
      inc() {
        this.$store.commit('JIA', this.n)
      },
      des() {
        this.$store.commit('JIAN', this.n)
      },
      incWhenJi() {
        this.$store.dispatch('jiaWhenJi', this.n)
      },
      incWait() {
        this.$store.dispatch('jiaWait', this.n)
      }
    }
  }
</script>

    6. 模块化开发

    store/index.js
    import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

const countOptions = {
  // 开启命名空间
  namespaced: true,
  actions: {
    jiaWhenJi(context, value) {
      console.log('actions中jiaWhenJi被调用')
      if (context.state.sum % 2) {
        context.commit('JIA', value)
      }
    },
    jiaWait(context, value) {
      console.log('actions中jiaWait被调用')
      setTimeout(() => {
        context.commit('JIA', value)
      }, 1000)
    }
  },
  mutations: {
    JIA(state, value) {
      console.log('mutations中JIA被调用')
      state.sum += value
    },
    JIAN(state, value) {
      console.log('mutations中JIAN被调用')
      state.sum -= value
    }
  },
  state: { sum: 0 },
  getters: {
    bigSum(state) {
      return state.sum * 10
    }
  }
}

const personOptions = {
  namespaced: true,
  actions: {},
  mutations: {
    ADD_PERSON(state, value) {
      console.log('mutations中ADD_PERSON被调用')
      state.personList.push(value)
    }
  },
  state: { personList: [] },
  getters: {}
}

// 创建并暴露store
export default new Vuex.Store({
  modules: {
    countAbout: countOptions,
    personAbout: personOptions
  }
})
    MyCount.vue
    <template>
  <div>
    <h1>sum: {{ sum }}</h1>
    <h1>扩大10倍后的sum: {{ bigSum }}</h1>
    <h1>personList个数: {{ personList.length }}</h1>
    <select v-model.number="n">
      <option value="1">1</option>
      <option value="2">2</option>
      <option value="3">3</option>
    </select>
    <!-- 如果不参数，则默认参数为点击事件 -->
    <button @click="inc(n)">加</button>
    <button @click="des(n)">减</button>
    <button @click="incWhenJi(n)">当sum为奇数时加</button>
    <button @click="incWait(n)">等等再加</button>
  </div>
</template>
<script>
  import { mapActions, mapGetters, mapMutations, mapState } from 'vuex'
  export default {
    name: 'MyCount',
    data() {
      return {
        n: 1
      }
    },
    mounted() {
      console.log(this)
    },
    computed: {
      ...mapState('personAbout', ['personList']),
      ...mapState('countAbout', ['sum']),

      ...mapGetters('countAbout', ['bigSum'])
    },
    methods: {
      ...mapActions('countAbout', { incWhenJi: 'jiaWhenJi', incWait: 'jiaWait' }),
      ...mapMutations('countAbout', { inc: 'JIA', des: 'JIAN' })
    }
  }
</script>
    MyPerson.vue
    <template>
  <div>
    <h1>personList个数: {{ personList.length }}</h1>
    <h1>sum: {{ sum }}</h1>
    <h1>sum: {{ this.$store.state.countAbout.sum }}</h1>
    <h1>bigSum: {{ bigSum }}</h1>
    <h1>bigSum: {{ this.$store.getters['countAbout/bigSum'] }}</h1>
    <input type="text" placeholder="请输入名称" v-model="name" />
    <button @click="add">添加</button>
    <ul>
      <li v-for="p in personList" :key="p.id">{{ p.name }}</li>
    </ul>
  </div>
</template>
<script>
  import { nanoid } from 'nanoid' // npm install nanoid
  import { mapState } from 'vuex'
  export default {
    name: 'MyPerson',
    data() {
      return {
        name: ''
      }
    },
    computed: {
      ...mapState('personAbout', ['personList']),
      ...mapState('countAbout', ['sum']),
      bigSum() {
        return this.$store.getters['countAbout/bigSum']
      }
    },
    methods: {
      add() {
        this.$store.commit('personAbout/ADD_PERSON', { id: nanoid(), name: this.name })
        this.name = ''
      }
    }
  }
</script>
